# merge all anc output files to one file per species

# need to add anc node and species name to the file
for f in N*/refined_*_anc_breakpoints.bed; do
  node=$(basename $(dirname "$f") | cut -d'_' -f1)
  species=$(basename "$f" | sed 's/refined_//; s/_anc_breakpoints.bed//')
  awk -v node="$node" -v sp="$species" 'BEGIN{OFS="\t"} {print $0, node, sp}' "$f"
done | sort -k7 -Vk1 > combined_anc_breakpoints.bed
